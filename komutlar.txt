docker build -t my-docker-project . # build image
docker run -p 3000:3000 -d my-docker-project # run in port 3000 and in detached mode
docker ps # show running containers
docker logs 2240486aa0d3 # show logs for container<id>
docker logs -f # follow
docker stop 2240486aa0d3 # stop given container
docker ps -a # show all container including the stopped ones
docker rm -f 2240486aa0d3 # remove given container
docker-compose up --build # build and run services in docker-compose.yml file
docker-compose down # stop and remove containers
docker-compose down -v # stop and remove containers and volumes
docker stats # show resource usage of containers
